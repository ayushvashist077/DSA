#include<stdio.h>
#include<stdlib.h>
 struct node
 {
     int data;
     struct node *next;
 }*first,*last,*newn;
   
struct node* create(int x)
{    
    newn=(struct node*)malloc(sizeof(struct node));
     newn->data=x;
    newn->next=NULL;
    return newn;
}
void insert()
{    int n; 
 printf("enter the value of the node: ");
 scanf("%d",&n);
 newn=create(n);
 if(first==NULL&&last==NULL)
 {
    first=last=newn;
    first->next=last->next=NULL;
 }
  else if(first->next==NULL)
 {  
  first->next=newn;
  last=newn;
  last->next=NULL;
 }
 else
 {
  last->next=newn;
  last=newn;
 }
}
void duplicate(struct node *ptr)
{
     struct node *q=ptr->next;
     struct node *t=ptr;
     while(ptr->next!=NULL)
     {
         while(q!=NULL)
         {
           if(ptr->data==q->data)
           {
             t->next=q->next;
             free(q);
             q=t->next;
           }
           else
           {
             q=q->next;
             t=t->next;
           }
         }
         ptr=ptr->next;
         t=ptr;
         q=ptr->next;
     }

}
void display(struct node *ptr)
{
    while(ptr!=NULL)
    {
        printf(" %d ",ptr->data);
        ptr=ptr->next;
    }
    printf("\n");
}
int main()
{  int n=100;
  while(n!=0)
  {
   printf("1->TO INSERT\n2->FOR SORTING\n3->DISPLAY\n0->TO EXIT\n");
   scanf("%d",&n);
   switch(n)
   {
    case 1:
    insert();
    break;
    case 2:
    duplicate(first);
    break;
    case 3:
    display(first);
    break;
   }
 }
}
